00026     public:
00027         KQP_SCALAR_TYPEDEFS(Scalar);

00029         UnaryKernelSpace(const FSpace &base) :  m_base(base) {}

00031         ~UnaryKernelSpace() {}
00032         
00033         virtual FMatrixBasePtr newMatrix() const override { return m_base->newMatrix(); }
00034         virtual FMatrixBasePtr newMatrix(const FeatureMatrixBase<Scalar> &other) const override { return m_base->newMatrix(other); }        
00035 
00036         virtual const ScalarMatrix &k(const FMatrixBase &mX) const override {            
00037             ScalarMatrix &gram = m_gramCache[static_cast<const void*>(&mX)];
00038             
00039             if (mX.size() == gram.rows()) return gram;
00040             
00041             // We lose space here, could be used otherwise???
00042             Index current = gram.rows();
00043             if (current < mX.size()) 
00044                 gram.conservativeResize(mX.size(), mX.size());
00045             
00046             Index tofill = mX.size() - current;
00047 
00048             fillGram(gram, tofill, mX);
00049             
00050             gram.bottomLeftCorner(tofill, current).noalias() = gram.topRightCorner(current, tofill).adjoint().eval();
00051             
00052             return gram;
00053         }
00054         
00055 
00056     protected:
00062         virtual void fillGram(ScalarMatrix &gram, Index tofill, const FMatrixBase &mX) const = 0;
00063 
00065         FSpace m_base;
00066 
00067     private:              
00069         mutable boost::unordered_map<const void *, ScalarMatrix> m_gramCache;

00075     template<typename Scalar> class GaussianSpace : public UnaryKernelSpace<Scalar> {
00076     public:
00077         KQP_SCALAR_TYPEDEFS(Scalar);
00078         using UnaryKernelSpace<Scalar>::m_base;
00079         using UnaryKernelSpace<Scalar>::k;
00080 
00081         GaussianSpace(Real sigma, const FSpace &base) : UnaryKernelSpace<Scalar>(base), m_sigma(sigma) {}
00082         virtual FSpaceBasePtr copy() const override { return FSpaceBasePtr(new GaussianSpace<Scalar>(m_sigma, m_base)); }        
00083 
00084         virtual Index dimension() const override { return -1; }
00085         virtual bool canLinearlyCombine() const override { return false; }
00086 
00087         
00088         virtual ScalarMatrix k(const FMatrixBase &mX1, const ScalarAltMatrix &mY1, const RealAltVector &mD1, 
00089                                const FMatrixBase &mX2, const ScalarAltMatrix &mY2, const RealAltVector &mD2) const override {
00090             return mD1.asDiagonal() * mY1.transpose() 
00091                     * this->f(m_base->k(mX1, mX2), m_base->k(mX1).diagonal(), m_base->k(mX2).diagonal())
00092                     * mY2 * mD2.asDiagonal();
00093         }
00094         
00095     protected:
00096         virtual void fillGram(ScalarMatrix &gram, Index tofill, const FMatrixBase &mX) const override {
00097             // Compute the remaining inner products
00098             auto norms = gram.diagonal();
00099             
00100             gram.rightCols(tofill) = this->f(m_base->k(mX).rightCols(tofill), norms, norms.tail(tofill));            
00101         }
00102         
00103 
00104     private:
00105         
00106         template<typename Derived, typename DerivedRow, typename DerivedCol>
00107         inline ScalarMatrix f(const Eigen::MatrixBase<Derived> &k, 
00108                                   const Eigen::MatrixBase<DerivedRow>& rowNorms, 
00109                                   const Eigen::MatrixBase<DerivedCol>& colNorms) const { 
00110             return ((rowNorms.derived().rowwise().replicate(k.cols()) + colNorms.derived().transpose().colwise().replicate(k.rows()) - 2 * k.derived()) / (m_sigma*m_sigma)).array().exp();
00111         }
00112       
00113         Real m_sigma;
00114     };

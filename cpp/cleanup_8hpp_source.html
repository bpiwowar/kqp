00072     template<typename Scalar>
00073     class StandardCleaner : public Cleaner<Scalar> {
00074     public:
00075         KQP_SCALAR_TYPEDEFS(Scalar);
00076         
00080         void cleanup(Decomposition<Scalar> &d) override {
00081             // --- Rank selection   
00082             DecompositionList<Real> list(d.mD);
00083             if (this->selector) {
00084                 
00085                 this->selector->selection(list);
00086                 
00087                 // Remove corresponding entries
00088                 select_rows(list.getSelected(), d.mD, d.mD);
00089                 
00090                 // Case where mY is the identity matrix
00091                 if (d.mY.getTypeId() == typeid(typename AltDense<Scalar>::IdentityType)) {
00092                     d.mX = d.mX.subset(list.getSelected());
00093                     d.mY.conservativeResize(list.getRank(), list.getRank());
00094                 } else {
00095                     select_columns(list.getSelected(), d.mY, d.mY);
00096                 }
00097             }
00098             
00099             // --- Remove unused pre-images
00100             RemoveUnusedPreImages<Scalar>::run(d.mX, d.mY);
00101             
00102             // --- Remove null space
00103             ReducedSetNullSpace<Scalar>::run(d.fs, d.mX, d.mY);
00104             
00105             // --- Ensure we have a small enough number of pre-images
00106             if (d.mX.size() > (this->preImageRatios.second * d.mD.rows())) {
00107                 if (d.fs.canLinearlyCombine() && this->useLinearCombination) {
00108                     // Easy case: we can linearly combine pre-images
00109                     d.mX = d.fs.linearCombination(d.mX, d.mY);
00110                     d.mY = ScalarMatrix::Identity(d.mX.size(), d.mX.size());
00111                 } else {
00112                     // Use QP approach
00113                     ReducedSetWithQP<Scalar> qp_rs;
00114                     qp_rs.run(this->preImageRatios.first * d.mD.rows(), d.fs, d.mX, d.mY, d.mD);
00115                     
00116                     // Get the decomposition
00117                     d.mX = qp_rs.getFeatureMatrix();
00118                     d.mY = qp_rs.getMixtureMatrix();
00119                     d.mD = qp_rs.getEigenValues();
00120                     
00121                     // The decomposition is not orthonormal anymore
00122                     d.orthonormal = false;
00123                 }
00124                 
00125             }
00126         }
00127         
00128     };
00129 }

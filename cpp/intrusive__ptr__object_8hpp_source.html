00062 // objects are derived last from this class; this way they
00063 // can be reference counted
00064 class intrusive_ptr_object
00065 {
00066 public:
00067         intrusive_ptr_object() throw()
00068                 : m_ref_count(0)
00069         {
00070         }
00071         virtual ~intrusive_ptr_object() = 0;
00072 
00073         void add_ref() const
00074         {
00075 #if defined(BOOST_HAS_THREADS)
00076                 mutex::scoped_lock lock(m_mutex);
00077 #endif
00078                 ++m_ref_count;
00079         }
00080         void release() const
00081         {
00082 #if defined(BOOST_HAS_THREADS)
00083                 mutex::scoped_lock lock(m_mutex);
00084 #endif
00085                 // since users should not call release() themselves but use
00086                 // the intrusive_ptr<> template instead, this should never happen
00087                 BOOST_ASSERT(m_ref_count != 0);
00088                 --m_ref_count;
00089                 if(m_ref_count == 0) {
00090                         delete this;
00091                 }
00092         }
00093 
00094 private:
00095         mutable unsigned int    m_ref_count;
00096 #if defined(BOOST_HAS_THREADS)
00097         mutable mutex           m_mutex;
00098 #endif
00099 };

00102 inline intrusive_ptr_object::~intrusive_ptr_object()
00103 {
00104         // a stack object will be deleted without a release() call...
00105         // if you use objects properly, this should never happen!
00106         BOOST_ASSERT(m_ref_count == 0);

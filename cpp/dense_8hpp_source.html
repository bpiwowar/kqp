00142             if (m_matrix.cols() == m_gramMatrix.rows()) return m_gramMatrix;
00143             
00144             // We lose space here, could be used otherwise???
00145             Index current = m_gramMatrix.rows();

00147                 m_gramMatrix.conservativeResize(m_matrix.cols(), m_matrix.cols());

00152             m_gramMatrix.bottomRightCorner(tofill, tofill).noalias() = this->getMatrix().rightCols(tofill).adjoint() * this->getMatrix().rightCols(tofill);
00153             m_gramMatrix.topRightCorner(current, tofill).noalias() = this->getMatrix().leftCols(current).adjoint() * this->getMatrix().rightCols(tofill);
00154             m_gramMatrix.bottomLeftCorner(tofill, current) = m_gramMatrix.topRightCorner(current, tofill).adjoint().eval();
00155

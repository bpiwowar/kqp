00054         Sparse(const ScalarMatrix &mat, double threshold) : m_matrix(mat.rows(), mat.cols()) {            
00055             Matrix<Real, 1, Dynamic> thresholds = threshold * mat.colwise().norm();
00056 
00057             Matrix<Index, 1, Dynamic> countsPerCol((mat.array().abs() >= thresholds.colwise().replicate(mat.rows()).array()).template cast<Index>().colwise().sum());
00058             
00059             m_matrix.reserve(countsPerCol);
00060             
00061             for(Index i = 0; i < mat.rows(); i++)
00062                 for(Index j = 0; j < mat.cols(); j++)
00063                     if (std::abs(mat(i,j)) > thresholds[j]) 
00064                         m_matrix.insert(i,j) = mat(i,j);
00065         }

00097         const ScalarMatrix &gramMatrix() const {
00098             Index current = m_gramMatrix.rows();
00099             if (size() == current) return m_gramMatrix;
00100             
00101             if (current < size()) 
00102                 m_gramMatrix.conservativeResize(size(), size());
00103             Index tofill = size() - current;
00104             
00105             // Compute the remaining inner products
00106             Index ncols = m_matrix.cols();
00107             
00108             for(Index i = 0; i < ncols; ++i)
00109                 for(Index j = current; j < ncols; ++j)
00110                     m_gramMatrix(i,j) = m_matrix.col(i).dot(m_matrix.col(j));
00111             m_gramMatrix.bottomLeftCorner(tofill, current) = m_gramMatrix.topRightCorner(current, tofill).adjoint().eval();
00112             
00113             return m_gramMatrix;
00114         }

00117         FMatrixBasePtr subset(const std::vector<bool>::const_iterator &begin, const std::vector<bool>::const_iterator &end) const override{
00118             // Construct
00119             std::vector<Index> selected;
00120             auto it = begin;
00121             for(Index i = 0; i < m_matrix.cols(); i++) {
00122                 if (it == end || *it)
00123                     selected.push_back(i);
00124                 it++;
00125             }
00126             
00127             // Prepare the resultant sparse matrix
00128             Storage s(m_matrix.rows(), selected.size());
00129             Eigen::VectorXi counts(selected.size());
00130             for(size_t i = 0; i < selected.size(); ++i)
00131                 counts[i] = m_matrix.col(selected[i]).nonZeros();
00132             s.reserve(counts);
00133             
00134             // Fill the result
00135             for(size_t i = 0; i < selected.size(); ++i)
00136                 for (typename Storage::InnerIterator it(m_matrix,selected[i]); it; ++it) 
00137                     s.insert(it.row(), i) = it.value();
00138             
00139             return FMatrixBasePtr(new Self(std::move(s)));

00110         template<typename _Scalar> struct Merge<_Scalar, typename boost::enable_if<boost::is_complex<_Scalar> >::type> {
00111             static void merge(KernelEVD<_Scalar>  &merger, const Decomposition<Scalar> &d) {
00112                 ScalarMatrix mY = d.mY * d.mD.asDiagonal();
00113                 merger.add(1, d.mX, d.mY.cwiseSqrt());
00114             }

00117         template<typename _Scalar> struct Merge<_Scalar, typename boost::disable_if<boost::is_complex<_Scalar> >::type> {
00118             static void merge(KernelEVD<_Scalar>  &merger, const Decomposition<Scalar> &d) {
00119 
00120                 Index posCount = 0;
00121 
00122                 for(Index j = 0; j <  d.mD.size(); j++) 
00123                     if (d.mD(j,0) >= 0) posCount++;
00124                 Index negCount = d.mD.size() - posCount;
00125                 
00126                 // FIXME: block expression for Alt expression
00127                 ScalarMatrix mY = d.mY * d.mD.cwiseAbs().cwiseSqrt().asDiagonal();
00128                 
00129                 Index jPos = 0;                
00130                 ScalarMatrix _mYPos(mY.rows(), posCount);
00131                 for(Index j = 0; j <  d.mD.size(); j++) 
00132                      if (d.mD(j,0) >= 0)
00133                          _mYPos.col(jPos++) = mY.col(j);
00134                 ScalarMatrix mYPos;
00135                 mYPos.swap(_mYPos);
00136                 
00137                 jPos = 0;
00138                 ScalarMatrix _mYNeg(mY.rows(), negCount);
00139                 for(Index j = 0; j <  d.mD.size(); j++) 
00140                     if (d.mD(j,0) < 0)
00141                         _mYPos.col(jPos++) = mY.col(j);
00142                 ScalarAltMatrix mYNeg;
00143                 mYNeg.swap(_mYNeg);
00144                 
00145                 if (posCount > 0) 
00146                     merger.add(1, d.mX, mYPos);
00147                 if (negCount > 0) 
00148                     merger.add(-1, d.mX, mYNeg);
00149             }

00157         void merge(bool force) {
00158             // Merge while the number of merged decompositions is the same for the two last decompositions
00159             // (or less, to handle the case of previous unbalanced merges)
00160             while (decompositions.size() >= 2 && (force || (decompositions.back().updateCount >= (decompositions.end()-1)->updateCount))) {
00161                 merger->reset();
00162                 for(int i = 0; i < 2; i++) {
00163                     const Decomposition<Scalar> &d = decompositions.back();
00164                     Merge<Scalar>::merge(*merger, d);
00165                     decompositions.pop_back();
00166                 } 
00167                 
00168                 // Push back new decomposition
00169                 decompositions.push_back(merger->getDecomposition());
00170                 if (mergerCleaner.get())
00171                     mergerCleaner->cleanup(decompositions.back());
00172             }
00173         }

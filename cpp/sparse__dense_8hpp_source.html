00061             // --- Compute which rows we need
00062             
00063

00065             RealVector norms(mat.cols());

00070                     norms[k] += Eigen::internal::abs2(it.value());

00072             }

00317         FMatrixBasePtr linearCombination(const ScalarAltMatrix &mA, Scalar alpha, const Self *mY, const ScalarAltMatrix *mB, Scalar beta) const {
00318             // Simple case: we don't have to add another matrix
00319             if (!mY) 
00320                 return FMatrixBasePtr(new Self(RowMap(m_map), alpha * m_matrix * mA));
00321             
00322             // Add the keys
00323             RowMap newMap;
00324             std::set_union(mY->m_map.begin(), mY->m_map.end(), m_map.begin(), m_map.end(), 
00325                            boost::make_function_output_iterator(Insert({newMap})));
00326 
00327 
00328             // Perform the linear combination
00329             ScalarMatrix mat(newMap.size(), mA.cols());
00330             mat.setZero();
00331             
00332             for(auto i = m_map.begin(), end = m_map.end(); i != end; i++) 
00333                 mat.row(newMap[i->first]) = alpha * m_matrix.row(i->second) * mA;
00334 
00335             for(auto i = mY->m_map.begin(), end = mY->m_map.end(); i != end; i++) 
00336                 mat.row(newMap[i->first]) += beta * mY->m_matrix.row(i->second) * *mB;
00337             
00338             // Move and cleanup before returning
00339             FMatrixBasePtr sdMat(new Self(std::move(newMap), std::move(mat)));
00340             dynamic_cast<Self&>(*sdMat).cleanup(EPSILON);

00041     template <typename Scalar> class AccumulatorKernelEVD<Scalar, true> : public KernelEVD<Scalar>  {
00042     public:
00043         enum {
00044             use_linear_combination = 1
00045         };
00046         
00047         KQP_SCALAR_TYPEDEFS(Scalar);
00048         
00049         AccumulatorKernelEVD(const FSpace &fs) : KernelEVD<Scalar>(fs), fMatrix(fs.newMatrix()) {
00050         }
00051         
00052         virtual ~AccumulatorKernelEVD() {
00053         }
00054         
00055         
00056         virtual void _add(Real alpha, const FMatrix &mX, const ScalarAltMatrix &mA) override {           
00057             // Just add the vectors using linear combination
00058             FMatrix fm = this->getFSpace().linearCombination(mX, mA, Eigen::internal::sqrt(alpha));
00059             fMatrix->add(*fm);
00060         }
00061         
00062         void reset() override {
00063             *this = AccumulatorKernelEVD(this->getFSpace());
00064         }
00065         
00067         virtual Decomposition<Scalar> _getDecomposition() const override {
00068             
00069             const ScalarMatrix& gram = this->getFSpace().k(fMatrix);
00070             Eigen::SelfAdjointEigenSolver<ScalarMatrix> evd(gram.template selfadjointView<Eigen::Lower>());
00071             
00072             ScalarMatrix _mY;
00073             RealVector _mD;
00074             kqp::thinEVD(evd, _mY, _mD);
00075             
00076             ScalarMatrix __mY;
00077             __mY.noalias() = _mY * _mD.cwiseAbs().cwiseSqrt().cwiseInverse().asDiagonal();
00078             
00079             return Decomposition<Scalar>(this->getFSpace(), fMatrix, __mY, _mD, true);
00080         }
00081         
00082     private:
00084         FMatrix fMatrix;

00091     template <typename Scalar> class AccumulatorKernelEVD<Scalar, false> : public KernelEVD<Scalar>  {
00092     public:
00093 
00094         KQP_SCALAR_TYPEDEFS(Scalar);
00095         
00096         AccumulatorKernelEVD(const FSpace &fs) : KernelEVD<Scalar>(fs), fMatrix(fs.newMatrix()) {
00097             offsets_X.push_back(0);
00098             offsets_A.push_back(0);
00099         }
00100         
00101         virtual ~AccumulatorKernelEVD() {}
00102         
00103     protected:
00104         virtual void _add(Real alpha, const FMatrix &mX, const ScalarAltMatrix &mA) override {           
00105             // If there is nothing to add            
00106             if (mA.cols() == 0)
00107                 return;
00108             
00109             // Do a deep copy of mA
00110             combination_matrices.push_back(mA);
00111             
00112             alphas.push_back(Eigen::internal::sqrt(alpha));
00113             fMatrix.add(mX);
00114             offsets_X.push_back(offsets_X.back() + mX.size());
00115             offsets_A.push_back(offsets_A.back() + mA.cols());
00116         }
00117         
00118         void reset() {
00119             *this = AccumulatorKernelEVD(this->getFSpace());
00120         }
00121 
00122         
00123     protected:
00125         virtual Decomposition<Scalar> _getDecomposition() const override {
00126             Decomposition<Scalar> d(this->getFSpace());
00127             // Compute A^T X^T X A^T 
00128             // where A = diag(A_1 ... A_n) and X = (X_1 ... X_n)
00129             
00130             Index size = offsets_A.back();
00131             
00132             // Nothing to do
00133             if (size == 0) {
00134                 d.mX = this->getFSpace().newMatrix();
00135                 d.mY.resize(0,0);
00136                 d.mD.resize(0,1);
00137                 return d;
00138             }
00139             
00140             ScalarMatrix gram_X = this->getFSpace().k(fMatrix);
00141             ScalarMatrix gram(size, size);
00142             
00143             for(size_t i = 0; i < combination_matrices.size(); i++) {
00144                 const ScalarAltMatrix &mAi = combination_matrices[i];
00145                 for(size_t j = 0; j <= i; j++) {
00146                     const ScalarAltMatrix &mAj = combination_matrices[j];
00147                     getBlock(gram, offsets_A, i, j) 
00148                             =  (mAi.transpose() *  ((Eigen::internal::conj(alphas[i]) * alphas[j]) * getBlock(gram_X, offsets_X, i, j))) * mAj;
00149                 }
00150             }
00151             
00152             
00153             // Direct EVD
00154             ScalarMatrix _mY;
00155             RealVector _mD;
00156             Eigen::SelfAdjointEigenSolver<ScalarMatrix> evd(gram.template selfadjointView<Eigen::Lower>());
00157             kqp::thinEVD(evd, _mY, _mD);
00158             d.mD.swap(_mD);
00159             
00160             // Y <- A * Y * D^-1/2
00161             
00162             ScalarMatrix _mY2 = _mY * d.mD.cwiseSqrt().cwiseAbs().cwiseInverse().asDiagonal();
00163             _mY = _mY2;
00164             ScalarMatrix __mY(offsets_X.back(), _mY.cols());
00165             
00166             for(size_t i = 0; i < combination_matrices.size(); i++) {
00167                 const ScalarAltMatrix &mAi = combination_matrices[i];
00168                 __mY.block(offsets_X[i], 0, offsets_X[i+1]-offsets_X[i], __mY.cols()) = mAi * (alphas[i] * _mY.block(offsets_A[i], 0,  offsets_A[i+1]-offsets_A[i], _mY.cols()));
00169             }
00170             
00171             d.mY.swap(__mY);
00172             d.mX = fMatrix;
00173             return d;
00174         }
00175         
00176     private:
00177         static inline Eigen::Block<ScalarMatrix> getBlock(ScalarMatrix &m, const std::vector<Index> &offsets, size_t i, size_t j) {
00178             return m.block(offsets[i], offsets[j], offsets[i+1] - offsets[i], offsets[j+1]-offsets[j]);
00179         }

00183         
00185         std::vector<ScalarAltMatrix> combination_matrices;
00186         
00188         std::vector<Index> offsets_A;
00189         std::vector<Index> offsets_X;
00190         
00191         
00193         std::vector<Scalar> alphas;
00194

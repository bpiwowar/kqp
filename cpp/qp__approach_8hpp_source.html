00053     struct LambdaError {
00054         Scalar delta;
00055         Scalar maxa;
00056         Index j;
00057         
00058         LambdaError() : delta(0), maxa(0) {}
00059         LambdaError(Scalar delta, Scalar maxa, Index j) : delta(delta), maxa(maxa), j(j) {}
00060         
00061         LambdaError operator+(const LambdaError &other) {
00062             LambdaError<Scalar> r;
00063             r.delta = this->delta + other.delta;
00064             r.maxa = this->maxa + other.maxa;
00065 //            std::cerr << "[[" << *this << " + " << other << "]] = " << r << std::endl;
00066             return r;
00067         }
00068         
00069         struct Comparator {
00070             bool operator() (const LambdaError &e1, const LambdaError &e2) { 
00071                 if (e1.maxa < EPSILON) 
00072                     return e1.maxa < e2.maxa;
00073                 return e1.delta * e2.maxa < e2.delta * e1.maxa; 
00074             }
00075         };
00076         
00077     };

00128         void run(Index target, const FSpace &fs, const FMatrix &mF, const ScalarAltMatrix &mY, const RealVector &mD) {            
00129             // Get the Gram matrix
00130             const ScalarMatrix &gram = fs.k(mF);
00131             
00132             // Dimension of the basis
00133             Index r = mY.cols();
00134 
00135             // Number of pre-images
00136             Index n = gram.rows();
00137             
00138             KQP_LOG_ASSERT_F(main_logger, mY.rows() == n, "Incompatible dimensions (%d vs %d)", %mY.rows() %n);
00139             
00140             //
00141             // (1) Estimate the regularization coefficient \f$lambda\f$
00142             //
00143 
00144            
00145             // mAS_.j = sigma_j^1/2 A_.j
00146             ScalarMatrix mAS = mY * mD.cwiseAbs().cwiseSqrt().asDiagonal();
00147            
00148             
00149             std::vector< LambdaError<Real> > errors;
00150             for(Index i = 0; i < n; i++) {
00151                 
00152                 Real maxa = mAS.cwiseAbs().row(i).maxCoeff();
00153                 Real delta = Eigen::internal::abs2(gram(i,i)) * mAS.cwiseAbs2().row(i).dot(mD.cwiseAbs().cwiseSqrt());
00154                 KQP_HLOG_DEBUG_F("[%d] We have max = %.3g/ delta = %.3g", %i %maxa %delta);
00155                 errors.push_back(LambdaError<Real>(delta, maxa, i));
00156             }
00157             
00158             
00159             typedef typename LambdaError<Real>::Comparator LambdaComparator;
00160             std::sort(errors.begin(), errors.end(), LambdaComparator());
00161             LambdaError<Real> acc_lambda = std::accumulate(errors.begin(), errors.begin() + n - target, LambdaError<Real>());
00162             
00163 //            for(Index j = 0; j < n; j++) 
00164 //                std::cerr << boost::format("[%d] delta=%g and maxa=%g\n") %j %errors[j].delta %errors[j].maxa;
00165 //
00166 //            std::cerr << boost::format("delta=%g and maxa=%g\n") %acc_lambda.delta % acc_lambda.maxa;
00167 
00168             Real lambda =  acc_lambda.delta / acc_lambda.maxa;   
00169             if (acc_lambda.maxa <= EPSILON * acc_lambda.delta) {
00170                 KQP_HLOG_WARN("There are only trivial solutions, a call to cleanUnused would have been better");
00171                 lambda = 1;
00172             }
00173                 
00174             KQP_HLOG_INFO_F("Lambda = %g", %lambda);                
00175             
00176             //
00177             // (2) Solve the cone quadratic problem
00178             //
00179             kqp::cvxopt::ConeQPReturn<Real> result;
00180             solve_qp<Scalar>(r, lambda, gram, mAS, mD.cwiseAbs().cwiseInverse().cwiseSqrt(), result);
00181 
00182             if (result.status == cvxopt::SINGULAR_KKT_MATRIX) 
00183                 KQP_THROW_EXCEPTION_F(arithmetic_exception, "QP approach did not converge (singular KKT matrix)", %result.status);
00184             
00185             // FIXME: case not converged
00186             
00187             //
00188             // (3) Get the subset 
00189             //
00190             
00191             // In result.x, the last n components are the maximum of the norm of the coefficients for each pre-image
00192             // we use the <target> first
00193             std::vector<Index> indices(n);
00194             for(Index i = 0; i < n; i++)
00195                 indices[i] = i;
00196             
00197             // Sort by increasing order: we will keep only the target last vectors
00198             std::sort(indices.begin(), indices.end(), getIndirectComparator(result.x.tail(n)));
00199             
00200             Real lowest   = result.x.tail(n)[indices[0]];
00201             Real last_nsel = n-target-1 >= 0 ? result.x.tail(n)[indices[n-target-1]] : -1;
00202             Real first_sel = result.x.tail(n)[indices[n-target]];
00203             Real highest    = result.x.tail(n)[indices[n-1]];
00204             KQP_HLOG_INFO_F("Lambda values [%d/%d]: lowest=%g, last non selected=%g, first selected=%g, highest=%g [ratios %g and %g]", 
00205                             %target %n %lowest %last_nsel %first_sel %highest %(last_nsel/first_sel) %(last_nsel/highest));
00206             
00207             if (KQP_IS_DEBUG_ENABLED(KQP_HLOGGER)) {
00208                 for(Index i = 0; i < n; i++) {
00209                     KQP_HLOG_DEBUG_F(boost::format("[%d] %g"), % indices[i] % result.x[result.x.size() - n + indices[i]]);
00210                 }
00211             }
00212             
00213             // Construct a sub-view of the initial set of indices
00214             std::vector<bool> to_keep(n, false);
00215             for(Index i = n-target; i < n; i++) {
00216                 to_keep[indices[i]] = true;
00217             }
00218             new_mF = mF.subset(to_keep.begin(), to_keep.end());
00219             
00220             
00221             //
00222             // (4) Project onto the new subspace
00223             //
00224             
00225             // Compute new_mY so that new_mF Y is orthonormal, ie new_mY' new_mF' new_mF new_mY is the identity
00226             
00227             Eigen::SelfAdjointEigenSolver<ScalarMatrix> evd(fs.k(new_mF).template selfadjointView<Eigen::Lower>());
00228             new_mY.swap(evd.eigenvectors());
00229             new_mY *= evd.eigenvalues().cwiseSqrt().cwiseInverse().asDiagonal();
00230             
00231             // Project onto new_mF new_mY
00232             
00233             KQP_MATRIX(Scalar) inner;
00234             inner = fs.k(new_mF, mF);
00235             new_mY *= new_mY.adjoint() * inner * mY;
00236 
00237             // Diagonal matrix does not change
00238             new_mD = mD;
00239 
00240         }

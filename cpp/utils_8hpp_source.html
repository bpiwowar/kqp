00034     template <class Derived> 
00035     void thinEVD(const Eigen::SelfAdjointEigenSolver<Derived> &evd, 
00036                  Eigen::Matrix<typename Derived::Scalar, Dynamic, Dynamic> &eigenvectors, 
00037                  Eigen::Matrix<typename Eigen::NumTraits<typename Derived::Scalar>::Real, Dynamic, 1> &eigenvalues,
00038                  Eigen::Matrix<typename Derived::Scalar, Dynamic, Dynamic> *nullEigenvectors = nullptr
00039                  ) {
00040         typedef typename Derived::Scalar Scalar;
00041         typedef typename Eigen::NumTraits<Scalar>::Real Real;
00042         
00043         // We expect eigenvalues to be sorted by increasing order
00044         Index dimension = evd.eigenvectors().rows();
00045         
00046         const Eigen::Matrix<Real,Dynamic,1> &d = evd.eigenvalues();
00047         Real threshold = EPSILON * (Real)d.size();
00048         
00049         Index n = d.rows();
00050         Index negatives = 0, zeros = 0;
00051         
00052         for(Index i = 0; i < n; i++) {
00053             assert(i==0 || d[i-1] <= d[i]);
00054             if (-d[i] > threshold) negatives++;
00055             else if (d[i] < threshold) zeros++; 
00056             else break;
00057         }
00058         
00059         Index positives = n - negatives - zeros;
00060         
00061         eigenvalues.resize(positives+negatives);
00062         eigenvalues.head(negatives) = d.head(negatives);
00063         eigenvalues.tail(positives) = d.tail(positives);
00064         
00065         eigenvectors.resize(dimension, positives + negatives);
00066         eigenvectors.leftCols(negatives) = evd.eigenvectors().leftCols(negatives);
00067         eigenvectors.rightCols(positives) = evd.eigenvectors().rightCols(positives);
00068         
00069         if (nullEigenvectors) {
00070             *nullEigenvectors = evd.eigenvectors().block(0, negatives, dimension, zeros);
00071         }
00072     }

00076     template <class Derived> 
00077     void thinEVD(const Eigen::SelfAdjointEigenSolver<Derived> &evd, 
00078                  typename AltDense<typename Derived::Scalar>::type &eigenvectors, 
00079                  typename AltVector<typename Eigen::NumTraits<typename Derived::Scalar>::Real>::type &eigenvalues,
00080                  typename AltDense<typename Derived::Scalar>::type *nullEigenvectors = nullptr) {
00081         
00082         typedef Eigen::Matrix<typename Derived::Scalar, Dynamic, Dynamic> ScalarMatrix;
00083         
00084         ScalarMatrix _eigenvectors;
00085         Eigen::Matrix<typename Eigen::NumTraits<typename Derived::Scalar>::Real, Dynamic, 1> _eigenvalues;
00086         boost::scoped_ptr<ScalarMatrix> _nullEigenvectors;
00087         
00088         
00089         if (nullEigenvectors != nullptr) 
00090             _nullEigenvectors.reset(new ScalarMatrix());
00091         
00092         thinEVD(evd, _eigenvectors, _eigenvalues, _nullEigenvectors.get());
00093         
00094         eigenvectors.swap(_eigenvectors);
00095         eigenvalues.swap(_eigenvalues);
00096         if (nullEigenvectors != nullptr) 
00097             nullEigenvectors->swap(*_nullEigenvectors);
00098         
00099     }

00153             auto negatives = mD.array() < 0;
00154             RealVector _mD;

00157             Index n = negatives.sum();

00161                     if (negatives[j]) selection[j] = true;
00162                 
00163                 select_rows(selection, mD, _mD);

00165             }
00166             
00167             
00168             // Perform the EVD
00169             ScalarMatrix m = fs.k(mX, mY, RealAltVector(mD.cwiseAbs().cwiseSqrt()));
00170             
00171             Eigen::SelfAdjointEigenSolver<decltype(m)> evd(m.template selfadjointView<Eigen::Lower>());
00172             kqp::thinEVD(evd, mY, mD);                    
00173             mY = mY * mD.cwiseAbs().cwiseSqrt().cwiseInverse().asDiagonal();
00174             
00175             // Handles negative eigenvalues
00176             if (n > 0) {
00177                 m.resize(mD.rows(), mD.rows());
00178                 auto _m = m.template selfadjointView<Eigen::Lower>();
00179                 m = mD;
00180                 _m.rankUpdate(mY.transpose() * fs.k(mX) * _mY * _mD, -2);
00181                 ScalarMatrix mU;
00182                 thinEVD(Eigen::SelfAdjointEigenSolver<decltype(m)>(_m), mU, mD);
00183                 mD = evd.eigenvalues();

00185             }

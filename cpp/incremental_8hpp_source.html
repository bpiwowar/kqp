00063         virtual void _add(Real alpha, const FMatrix &mU, const ScalarAltMatrix &mA) override {
00064             // --- Info
00065             
00066 //            KQP_LOG_DEBUG_F(KQP_HLOGGER, "Dimensions: X [%d], Y [%dx%d], Z [%dx%d], D [%d], U [%d], A [%dx%d]", 
00067 //                           %mX.size() %mY.rows() %mY.cols() %mZ.rows() %mZ.cols() %mD.rows() %mU.size() %mA.rows() %mA.cols());
00068             
00069             // --- Pre-computations
00070             
00071             // Compute W = Y^T X^T
00072             
00073             ScalarMatrix mW =  getFSpace().k(mX, mY, mU, mA);
00074             
00075             // Compute V^T V
00076             ScalarMatrix vtv = getFSpace().k(mU, mA);
00077             vtv -= mW.adjoint() * mW;
00078             
00079             
00080             // (thin) eigen-value decomposition of V^T V
00081             Eigen::SelfAdjointEigenSolver<ScalarMatrix> evd(vtv);
00082             ScalarMatrix mQ;
00083             ScalarMatrix mQ0;
00084             RealVector mDQ;
00085             kqp::thinEVD(evd, mQ, mDQ, &mQ0);
00086 
00087             Index rank_Q = mQ.cols();             
00088             mDQ = mDQ.cwiseAbs().cwiseSqrt();
00089             for(Index i = 0; i < rank_Q; i++)
00090                 mQ.col(i) /= mDQ(i);
00091 
00092             // --- Update
00093            
00094             // m is the matrix [WQD^1/2 WQ_0; D^1/2 0 ] 
00095             ScalarMatrix m(mW.rows() + rank_Q, mW.cols());
00096                         
00097             m.topLeftCorner(mW.rows(), rank_Q) =  mW * mQ * mDQ.asDiagonal();
00098             m.topRightCorner(mW.rows(), mW.cols() - rank_Q) = mW * mQ0;                
00099             
00100             m.bottomLeftCorner(rank_Q, rank_Q) = mDQ.template cast<Scalar>().asDiagonal();
00101             m.bottomRightCorner(rank_Q, mW.cols() - rank_Q).setConstant(0) ;
00102             
00103 
00104             for(Index i = 0; i < m.cols(); i++) {
00105                 EvdUpdateResult<Scalar> result;
00106                 // Update
00107                 ScalarVector v(m.rows());
00108                 v.head(mZ.cols()) = mZ.adjoint() * m.col(i).head(mZ.cols());
00109                 v.tail(m.rows() - mZ.cols()) = m.col(i).tail(m.rows() - mZ.cols());
00110                 
00111                 
00112                 // Rank-1 update:
00113                 // For better accuracy, we don't decrease the rank yet using the selector,
00114                 // but this might be an option in the future (so as to improve speed)
00115                 evdRankOneUpdate.update(mD, alpha, v, false, 0, false, result, &mZ);
00116                 
00117                 // Take the new diagonal
00118                 mD = result.mD;
00119             }
00120             
00121             // Update X and Y if the rank has changed
00122             if (rank_Q > 0) {
00123                 // Add the pre-images from U
00124                 mX.add(mU);
00125                 
00126                 // Update mY
00127                 Index old_Y_rows = mY.rows();
00128                 Index old_Y_cols = mY.cols();
00129                 mY.conservativeResize(mY.rows() + mA.rows(), mY.cols() + mQ.cols());
00130 
00131                 mY.bottomRightCorner(mA.rows(), mQ.cols()) = mA * mQ;
00132                 mY.bottomLeftCorner(mA.rows(), old_Y_cols).setConstant(0);
00133                 if (old_Y_rows > 0)
00134                     mY.topRightCorner(old_Y_rows, mQ.cols()) = - mY.topLeftCorner(old_Y_rows, mW.rows()) * mW * mQ;
00135             }
00136                               
00137             // (4) Clean-up
00138             
00139 
00140 
00141             // --- Rank selection   
00142             DecompositionList<Real> list(mD);
00143             bool identityZ = false;
00144             if (this->selector) {
00145                 // Selects the eigenvalues
00146                 this->selector->selection(list);
00147                 
00148                 // Remove corresponding entries
00149                 select_rows(list.getSelected(), mD, mD);                
00150                 mY = mY * mZ;
00151                 select_columns(list.getSelected(), mY, mY);
00152 
00153                 identityZ = true;
00154             }
00155             
00156             
00157             // First, tries to remove unused pre-images images
00158             RemoveUnusedPreImages<Scalar>::run(mX, mY);
00159             
00160             // --- Ensure we have a small enough number of pre-images
00161             Index maxRank = this->preImageRatios.second * (float)mD.rows();
00162             if (mX.size() > maxRank) {
00163                 
00164                 // Get rid of Z
00165                 if (!identityZ) mY = mY * mZ;
00166                 identityZ = true;
00167 
00168                 // Try again to remove unused pre-images
00169                 RemoveUnusedPreImages<Scalar>::run(mX, mY);
00170                 KQP_LOG_DEBUG_F(KQP_HLOGGER, "Rank after unused pre-images algorithm: %d [%d]", %mY.rows() %maxRank);
00171 
00172                 // Try to remove null space pre-images
00173                 ReducedSetNullSpace<Scalar>::run(getFSpace(), mX, mY);
00174                 KQP_LOG_DEBUG_F(KQP_HLOGGER, "Rank after null space algorithm: %d [%d]", %mY.rows() %maxRank);
00175 
00176                 if (mX.size() > maxRank) {
00177                     if (getFSpace().canLinearlyCombine() && this->useLinearCombination) {
00178                         // Easy case: we can linearly combine pre-images
00179                         mX = getFSpace().linearCombination(mX, mY);
00180                         mY = ScalarMatrix::Identity(mX.size(), mX.size());
00181                     } else {
00182                         // Use QP approach
00183                         ReducedSetWithQP<Scalar> qp_rs;
00184                         qp_rs.run(this->preImageRatios.first * (float)mD.rows(), getFSpace(), mX, mY, mD);
00185                         
00186                         // Get the decomposition
00187                         mX = qp_rs.getFeatureMatrix();
00188                         mY = qp_rs.getMixtureMatrix();
00189                         mD = qp_rs.getEigenValues();
00190                         
00191                         // The decomposition is not orthonormal anymore
00192                         Orthonormalize<Scalar>::run(getFSpace(), mX, mY, mD);
00193                     }
00194                 }
00195             
00196             }
00197             
00198             if (identityZ) 
00199                 mZ = ScalarMatrix::Identity(mY.cols(), mY.cols());
00200             
00201         }

00203         // Gets the decomposition
00204         virtual Decomposition<Scalar> _getDecomposition() const override {
00205             Decomposition<Scalar> d(this->getFSpace());
00206             
00207             d.mX = this->mX;
00208             
00209             const_cast<ScalarMatrix&>(this->mY) = this->mY * this->mZ;
00210             const_cast<ScalarMatrix&>(this->mZ) = ScalarMatrix::Identity(mY.rows(), mY.rows());
00211             
00212             d.mY = ScalarAltMatrix(this->mY);
00213             
00214             d.mD = this->mD;
00215             
00216             return d;
00217         }

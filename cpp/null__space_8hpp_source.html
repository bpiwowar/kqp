00056         static void remove(FMatrix &mF, ScalarMatrix &kernel, Eigen::PermutationMatrix<Dynamic, Dynamic, Index>& mP, const RealVector &weights, double delta = 1e-4) {
00057             typedef typename Eigen::PermutationMatrix<Dynamic, Dynamic, Index> Permutation;
00058             
00059             // --- Look up at the indices to remove
00060             
00061             // Get the pre-images available for removal (one per vector of the null space, i.e. by columns in kernel)
00062             // Summarize the result per pre-image (number of vectors where it appears)
00063             Eigen::Matrix<int,Dynamic,1> num_vectors = (kernel.array().abs() > kernel.array().abs().colwise().maxCoeff().replicate(kernel.rows(), 1) * delta).template cast<int>().rowwise().sum();
00064             
00065             std::vector<Index> to_remove;
00066             for(Index i = 0; i < num_vectors.rows(); i++)
00067                 if (num_vectors[i] > 0) 
00068                     to_remove.push_back(i);
00069             
00070             // Sort        
00071             const Index pre_images_count = kernel.rows();
00072             const std::size_t remove_size = kernel.cols();
00073             const Index keep_size = mF.size() - remove_size;
00074             
00075             mP = Permutation(mF.size());
00076             mP.setIdentity();
00077             
00078             std::sort(to_remove.begin(), to_remove.end(), IndirectSort<RealVector>(weights));
00079             std::vector<bool> selection(mF.size(), true);
00080             std::vector<bool> used(remove_size, false);
00081             
00082             ScalarVector v;
00083             
00084             // --- Remove the vectors one by one (Pivoted Gauss)
00085             
00086             for(size_t index = 0;  index < remove_size; index++) {
00087                 // remove the ith pre-image
00088                 size_t i = to_remove[index];
00089                 
00090                 // Searching for the highest magnitude
00091                 Index j = -1;
00092                 Real max = 0;
00093                 for(Index k = 0; k < kernel.cols(); k++) {
00094                     Real x = kernel.array().abs()(i, k);
00095                     if (!used[k] && x > max) {
00096                         max = x;
00097                         j = k;
00098                     }
00099                 }
00100                 assert(j != -1);
00101                 used[j] = true;
00102                 
00103                 // Update permutation
00104                 selection[i] = false;            
00105                 mP.indices()(i) = j + keep_size;
00106                 
00107                 // Update the matrix
00108                 v = kernel.col(j) / kernel(i,j);
00109                 kernel.col(j) /= -kernel(i,j);
00110                 kernel(i,j) = 0;
00111                 
00112                 kernel = ((ScalarMatrix::Identity(pre_images_count, pre_images_count) 
00113                            - v * ScalarVector::Unit(pre_images_count, i).transpose()) * kernel).eval();
00114             }
00115             
00116             
00117             // --- Remove the vectors from mF and set the permutation matrix
00118             
00119             // Remove unuseful vectors
00120             mF = mF.subset(selection.begin(), selection.end());
00121             select_rows(selection.begin(), selection.end(), kernel, kernel);
00122             
00123             // Complete the permutation matrix
00124             Index count = 0;
00125             for(size_t index = 0; index < selection.size(); index++) 
00126                 if (selection[index]) {
00127                     mP.indices()(index) = count++;
00128                 }
00129             
00130         }

00139         static void run(const FSpace &fs, FMatrix &mF, ScalarAltMatrix &mY) {
00140             
00141             // Removes unused pre-images
00142             RemoveUnusedPreImages<Scalar>::run(mF, mY);
00143             
00144             // Dimension of the problem
00145             Index N = mY.rows();
00146             assert(N == mF.size());
00147             
00148             // LDL decomposition (stores the L^T matrix)
00149             Eigen::FullPivLU<ScalarMatrix> lu_decomposition(fs.k(mF));
00150             
00151             // Stop if full rank
00152             if (lu_decomposition.rank() == N) 
00153                 return;
00154             
00155             // Remove pre-images using the kernel
00156             ScalarMatrix kernel = lu_decomposition.kernel();
00157             RealVector weights = mY.rowwise().squaredNorm().array() * fs.k(mF).diagonal().array().abs();
00158             Eigen::PermutationMatrix<Dynamic, Dynamic, Index> mP;
00159             remove(mF, kernel, mP, weights);
00160             
00161             // Y <- (Id A) P Y
00162             ScalarMatrix mY2(mY);
00163             mY2= (mP * mY2).topRows(mF.size()) + kernel * (mP * mY2).bottomRows(mY.rows() - mF.size());
00164             mY.swap(mY2);
00165             
00166             // Removes unused pre-images
00167             RemoveUnusedPreImages<Scalar>::run(mF, mY);
00168         }
